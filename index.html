const express = require("express"); const fetch = (...args) => import('node-fetch').then(({default: fetch}) => fetch(...args)); const app = express(); app.use(express.json()); const API_URL = "https://script.google.com/macros/s/AKfycbwQrMddICJXl_83X8TLe4hTD0LBcrbwRRp5hMHj9GNr8pcM1Oqm0ZGxWtk4H-6dFUXc/exec"; app.use((req, res, next) => { res.setHeader("Access-Control-Allow-Origin", "*"); res.setHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS"); res.setHeader("Access-Control-Allow-Headers", "Content-Type"); next(); }); app.post("/api", async (req, res) => { try { const response = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(req.body) }); const data = await response.json(); res.json(data); } catch (err) { res.json({ error: err.message }); } }); app.listen(8888, () => console.log("Proxy running on http://localhost:8888/api") );
